    //计算的是可以达到的最大值，那么就找到滑动窗口的最小值
    //sum = 保存所有和
    //curValue = 保存滑动串口的总和
    //minValue = 最小值
    //sum - minValue 就是最大值
    public int maxScore(int[] cardPoints, int k) {
        int n = cardPoints.length;
        int sum = 0;
        for (int num : cardPoints) {
            sum += num;
        }

        int cur = 0, curValue = 0;
        // n - k
        for ( ; cur < n - k; cur++) {
            curValue += cardPoints[cur];
        }
        int minValue = curValue;
        for (int i = cur; i < n ; i++) {
            curValue = curValue + cardPoints[i] - cardPoints[i - cur];
            minValue = Math.min(minValue,curValue);
        }
        return sum - minValue;
    }
}
